function editVendorSelected(id){
  //HERE WE GET THE ICON TO BE REPLACED LATER ON FOR X-icon AND CHECK-icon
  const tdParentOfIcon = document.getElementById(`${id}-option-td`);

  //HERE WE DELETE THE EDIT ICON AND APPLY CONTENT BELOW
  tdParentOfIcon.innerHTML = `
      <div class="d-flex justify-content-between">
        <div onclick="(() => cancel(${id}, 'cancel'))()">
          <i class="fa-solid fa-xmark"></i>
        </div>
        <div onclick="(() => cancel(${id}, 'save'))()">
          <i  class="fa-solid fa-check"></i>
        </div>
      </div>`
  // const editIcon = tdParentOfIcon[0];

  //HERE WE LOCATE THE TR BELONGING TO ID-OPTION 
  const selected = document.getElementById(`${id}-option`);
  // HERE WE GET ALL CHILDREN WITHIN THE TR PARENT ELEMENT 
  const tds = selected.getElementsByTagName('td');

  //HERE WE GATHER ALL TDS AND MAKE THEM EDITABLE AND ALSO IF IT IS SELECT TYPE, IT WILL BE ENABLED
  for(let i = 0; i < tds.length -2; i++){
    tds[i].setAttribute('contenteditable', 'true')
    if(tds[i + 1].getElementsByTagName('select')[0]){
      tds[i + 1].getElementsByTagName('select')[0].removeAttribute('disabled')
    } 
  }

  // const cancel = tdParentOfIcon.querySelector(".fa-xmark");
  // cancel.addEventListener('click', function(){
  //   tdParentOfIcon.innerHTML = `<i class="fa-solid fa-user-pen"></i>`;
  // });
}
